        \begin{align*}
                u_a &= \frac{R_4 (R_1 + R_3)}{R_1 (R_2 + R_4)} u_{e2} - \frac{R_3}{R_1} u_{e1} \\
                u_a &= \frac{R_3}{R_1} (u_{e2} - u_{e1}) \qquad \text{wenn} \quad \frac{R_1}{R_3} = \frac{R_2}{R_4}
        \end{align*}
